// Resume from last checkpoint function resumeFromCheckpoint() { gameOver = false; ufo.y = GAME_HEIGHT/2; ufo.vel = 0; score = lastCheckpointScore; coinsCollected = lastCheckpointCoins; pipes = []; coins = []; pipeCount = lastCheckpointScore; // approximate pipe count for checkpoint } // Reset game from start function resetGame() { gameOver = false; ufo.y = GAME_HEIGHT/2; ufo.vel = 0; score = 0; coinsCollected = 0; lastCheckpointScore = 0; lastCheckpointCoins = 0; pipes = []; coins = []; pipeCount = 0; } // Update game function update() { ctx.save(); ctx.scale(scaleX, scaleY); ctx.clearRect(0, 0, GAME_WIDTH, GAME_HEIGHT); if(!gameOver) ufo.vel += ufo.gravity; ufo.y += ufo.vel; // Boundaries if(ufo.y + ufo.h > GAME_HEIGHT || ufo.y < 0) gameOver = true; // Draw UFO ctx.drawImage(ufoImg, ufo.x, ufo.y, ufo.w, ufo.h); // Add pipes if(pipes.length === 0 || pipes[pipes.length-1].x < GAME_WIDTH - 300) addPipe(); // Draw pipes pipes.forEach((p,i)=>{ if(!gameOver) p.x -= 3; ctx.fillStyle = 'green'; ctx.fillRect(p.x, 0, pipeWidth, p.top); ctx.fillRect(p.x, GAME_HEIGHT - p.bottom, pipeWidth, p.bottom); // Score if(!p.passed && ufo.x > p.x + pipeWidth){ p.passed = true; score++; if(p.checkpoint){ lastCheckpointScore = score; lastCheckpointCoins = coinsCollected; } } if(collision(p)) gameOver = true; if(p.x + pipeWidth < 0) pipes.splice(i,1); }); // Draw coins coins.forEach((c,i)=>{ if(!gameOver) c.x -=3; if(collectCoin(c)){ c.collected = true; coinsCollected++; if(pipes.find(p=>p.checkpoint && c.x >= p.x && c.x <= p.x+pipeWidth)){ lastCheckpointScore = score; lastCheckpointCoins = coinsCollected; } } if(!c.collected) ctx.drawImage(coinImg, c.x, c.y, c.w, c.h); if(c.x + c.w < 0) coins.splice(i,1); }); // Draw score and coins ctx.fillStyle = 'white'; ctx.font = '30px Arial'; ctx.fillText(`Score: ${score}`, 20, 50); ctx.fillText(`Coins: ${coinsCollected}`, 20, 90); // Game over if(gameOver){ ctx.fillStyle = 'red'; ctx.font = '50px Arial'; ctx.fillText('🛸 GAME OVER 🛸', GAME_WIDTH/2 - 180, GAME_HEIGHT/2 - 30); ctx.font = '30px Arial'; ctx.fillText('Tap anywhere to return to last checkpoint', GAME_WIDTH/2 - 250, GAME_HEIGHT/2 + 20); } // Draw checkpoint text pipes.forEach(p=>{ if(p.checkpoint){ ctx.fillStyle = 'yellow'; ctx.font = '25px Arial'; ctx.fillText('Checkpoint', p.x + 5, p.top + pipeGap/2 + 10); } }); ctx.restore(); requestAnimationFrame(update); } // Handle tap function flap(event){ let rect = canvas.getBoundingClientRect(); let x = (event.clientX || event.touches[0].clientX) - rect.left; let y = (event.clientY || event.touches[0].clientY) - rect.top; x /= scaleX; y /= scaleY; if(gameOver) resumeFromCheckpoint(); else ufo.vel = ufo.lift; } document.addEventListener('touchstart', flap); document.addEventListener('mousedown', flap); document.getElementById('resetBtn').addEventListener('click', resetGame); update();
